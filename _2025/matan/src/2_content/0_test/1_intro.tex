

\hlTerm{Множество}{fletter=A; B; X; ... (любая большая буква)} –-- совокупность объектов любой природы.\\

\hlTerm{Элементы множества}{fletter=a; b; x; ... (любая маленькая буква)} –-- Объекты, составляющие данное множество.\\


\begin{Example}{}
    Пример множества состоящего из элементов x_1, x_2, ..., x_n:

    $$ \{x_1, x_2, ..., x_n\} $$
\end{Example}

\begin{Example}{}
    Пример принадлежности элемента x к множеству X:

    \begin{RBox}{}
        gh for h
        % $ x \in X $
    \end{RBox}

    \begin{ExampleTask}{}
        $ x \in X $
    \end{ExampleTask}

\end{Example}

\hlTerm{Подмножество множества}{} –-- Если любой элемент множества Х является
элементом множества Y, то множество Х называется подмножеством множества Y.\\
